    Future<List<T>> toList() {
  List<T> result = <T>[];
  _Future<List<T>> future = new _Future<List<T>>();
  this.listen(
    (T data) {
      result.add(data);
    },
    onError: future._completeError,
    onDone: () {
      future._complete(result);
    },
    cancelOnError: true);
  return future;
}
