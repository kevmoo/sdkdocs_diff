    factory TextEvent(String type,
  {bool canBubble: false, bool cancelable: false, Window view, String data}) {
  if (view == null) {
    view = window;
  }
  var e = document._createEvent("TextEvent");
  e._initTextEvent(type, canBubble, cancelable, view, data);
  return e;
}
