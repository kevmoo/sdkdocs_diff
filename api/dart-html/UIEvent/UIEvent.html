    factory UIEvent(String type,
    {Window view, int detail: 0, bool canBubble: true,
    bool cancelable: true}) {
  if (view == null) {
    view = window;
  }
  final e = document._createEvent("UIEvent");
  e._initUIEvent(type, canBubble, cancelable, view, detail);
  return e;
}
