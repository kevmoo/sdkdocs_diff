    static void register(Metric metric) {
  if (metric is! Metric) {
    throw new ArgumentError('metric must be a Metric');
  }
  if (_metrics[metric.name] != null) {
    throw new ArgumentError('Registered metrics have unique names');
  }
  _metrics[metric.name] = metric;
}
