    String toString() {
  StringBuffer sb = new StringBuffer();
  sb.write(type);
  if (!message.isEmpty) {
    sb.write(": $message");
    if (osError != null) {
      sb.write(" ($osError)");
    }
  } else if (osError != null) {
    sb.write(": $osError");
  }
  return sb.toString();
}
