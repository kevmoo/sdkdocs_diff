    String toString() {
  StringBuffer sb = new StringBuffer();
  sb.write("OS Error");
  if (!message.isEmpty) {
    sb..write(": ")
      ..write(message);
    if (errorCode != noErrorCode) {
      sb..write(", errno = ")
        ..write(errorCode.toString());
    }
  } else if (errorCode != noErrorCode) {
    sb..write(": errno = ")
      ..write(errorCode.toString());
  }
  return sb.toString();
}
