    @DomName('IDBObjectStore.put')
Future put(value, [key]) {
  try {
    var request;
    if (key != null) {
      request = _put(value, key);
    } else {
      request = _put(value);
    }
    return _completeRequest(request);
  } catch (e, stacktrace) {
    return new Future.error(e, stacktrace);
  }
}
